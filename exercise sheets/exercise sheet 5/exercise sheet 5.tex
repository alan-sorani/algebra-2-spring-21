\begin{exercise}%2
יהי
$V$
מרחב וקטורי מעל
$\mbb{C}$,
תהי
$T \in \endo_{\mbb{C}}\prs{V}$
ויהי
$W \lneq V$
תת־מרחב
$T$%
־שמור.
הוכיחו כי קיים
$\lambda \in \mbb{C}$
ערך עצמי של 
$T$
וקיים
$v \in V \setminus W$
עבורם מתקיים
$\prs{T-\lambda \id_V} v \in W$.
\end{exercise}

\begin{exercise}%3
\begin{enumerate}
\item מצאו את צורת ז'ורדן של
$J_n\prs{\lambda}^{-1}$
עבור
$\lambda \in \mbb{C}\setminus\set{0}$
ועבור
$n \in \mbb{N}_+$.

\item תהי
$A \in M_n\prs{\mbb{C}}$
הפיכה.
מצאו את צורת ז'ורדן של
$A^{-1}$.

\item מצאו תנאי הכרחי ומספיק על
$A \in M_n\prs{\mbb{C}}$
כך שיתקיים
$A \sim A^{-1}$.
\end{enumerate}
\end{exercise}
